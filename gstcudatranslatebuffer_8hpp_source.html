144 template <typename GstObjPtr>

146  GstCudaData *gst_buffer,

152  const rr::Format format =

156  const int width = gst_buffer->channels[0].width;
157  const int height = gst_buffer->channels[0].height;
158 
159  for (int i = 0; i < planes; ++i) {
160  strides[i] = gst_buffer->channels[i].pitch;
161  }
162

164
